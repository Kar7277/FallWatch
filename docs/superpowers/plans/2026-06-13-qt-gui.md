# EdgeWatcher QT 界面 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 开发 EdgeWatcher QT5 Widgets 上位机界面，支持 PC 远程监看（B版）和板子本地运行（A版）两种模式。

**Architecture:** QT 主窗口通过 `IDataProvider` 抽象接口获取数据，`RemoteDataProvider`（PC版）通过 TCP+MQTT 与板子通信，`LocalDataProvider`（板子版）通过共享内存获取帧和事件。C 后端新增 `shm_output` 和 `tcp_stream` 两个模块。

**Tech Stack:** Qt 5 Widgets, qmake, C++11, POSIX shared memory (shm_open/mmap), TCP sockets, paho-mqtt-c, libjpeg

```cpp
#include <QApplication>
#include "mainwindow.h"
#include "remotedataprovider.h"
#include "localdataprovider.h"

int main(int argc, char *argv[])
{
    QApplication app(argc, argv);

    // 解析启动参数选择数据源
    IDataProvider *provider = nullptr;
    bool localMode = false;

    for (int i = 1; i < argc; i++) {
        if (QString(argv[i]) == "--local")
            localMode = true;
    }

    if (localMode) {
        provider = new LocalDataProvider();
    } else {
        // B版默认：连接板子
        QString host = "192.168.1.100";
        int port = 9000;
        for (int i = 1; i < argc; i++) {
            if (QString(argv[i]) == "--host" && i+1 < argc)
                host = argv[++i];
            if (QString(argv[i]) == "--port" && i+1 < argc)
                port = QByteArray(argv[++i]).toInt();
        }
        provider = new RemoteDataProvider(host, port);
    }

    if (!provider->connect()) {
        qCritical("Failed to connect to data source");
        return 1;
    }

    MainWindow w(provider);
    w.show();

    return app.exec();
}
```

```cpp
    m_onColor = color;
    update();
}

void StatusDot::setOffColor(const QColor &color)

    m_offColor = color;
    update();

void StatusDot::setState(State state)
{
    m_state = state;
    if (state == Blinking) {
        m_blinkTimer.start(500);  // 500ms 交替
    } else {
        m_blinkTimer.stop();
        m_lit = (state == On);
    }
    update();
}

StatusDot::State StatusDot::state() const
{
    return m_state;
}

void StatusDot::paintEvent(QPaintEvent *)
{
    QPainter painter(this);
    painter.setRenderHint(QPainter::Antialiasing, true);

    bool drawLit = false;
    switch (m_state) {
    case On:       drawLit = true; break;
    case Off:      drawLit = false; break;
    case Blinking: drawLit = m_lit; break;
    }

    QColor color = drawLit ? m_onColor : m_offColor;
    painter.setBrush(color);
    painter.setPen(Qt::NoPen);
    painter.drawEllipse(rect().adjusted(2, 2, -2, -2));
}
```

```cpp
#include "remotedataprovider.h"
#include <QDataStream>
#include <QDebug>

RemoteDataProvider::RemoteDataProvider(const QString &host, int port)
    : m_host(host), m_port(port)
{
}

RemoteDataProvider::~RemoteDataProvider()
{
    disconnect();
}

bool RemoteDataProvider::connect()
{
    m_socket = new QTcpSocket(this);
    connect(m_socket, &QTcpSocket::readyRead,
            this, &RemoteDataProvider::onTcpReadyRead);
    connect(m_socket,
            QOverload<QAbstractSocket::SocketError>::of(&QAbstractSocket::error),
            this, &RemoteDataProvider::onTcpError);

    m_socket->connectToHost(m_host, m_port);
    if (!m_socket->waitForConnected(3000)) {
        qCritical("TCP connect to %s:%d failed: %s",
                  qPrintable(m_host), m_port,
                  qPrintable(m_socket->errorString()));
        return false;
    }

    qDebug("Connected to EdgeWatcher at %s:%d",
           qPrintable(m_host), m_port);
    return true;
}

void RemoteDataProvider::disconnect()
{
    if (m_socket) {
        m_socket->disconnectFromHost();
        m_socket->deleteLater();
        m_socket = nullptr;
    }
}

// TCP协议：4字节大端长度 + JPEG数据
void RemoteDataProvider::onTcpReadyRead()
{
    m_frameBuffer.append(m_socket->readAll());

    while (true) {
        // 还没读到长度头
        if (m_expectedLen == 0 && m_frameBuffer.size() < 4)
            return;

        // 解析长度
        if (m_expectedLen == 0) {
            QDataStream ds(m_frameBuffer.left(4));
            ds.setByteOrder(QDataStream::BigEndian);
            ds >> m_expectedLen;
            m_frameBuffer.remove(0, 4);
        }

        // 数据没收完
        if (m_frameBuffer.size() < m_expectedLen)
            return;

        // 收完一帧
        QByteArray jpegData = m_frameBuffer.left(m_expectedLen);
        m_frameBuffer.remove(0, m_expectedLen);
        m_expectedLen = 0;

        QImage frame;
        if (frame.loadFromData(jpegData, "JPEG")) {
            QMutexLocker lock(&m_frameMutex);
            m_lastFrame = frame;
        }
    }
}

void RemoteDataProvider::onTcpError(QAbstractSocket::SocketError)
{
    qWarning("TCP socket error: %s",
             qPrintable(m_socket->errorString()));
}

QImage RemoteDataProvider::getPreviewFrame()
{
    QMutexLocker lock(&m_frameMutex);
    return m_lastFrame;
}

int RemoteDataProvider::getWatcherStatus()
{
    QMutexLocker lock(&m_statusMutex);
    return m_watcherStatus;
}

int RemoteDataProvider::getMotionStatus()
{
    QMutexLocker lock(&m_statusMutex);
    return m_motionStatus;
}

int RemoteDataProvider::getWarningStatus()
{
    QMutexLocker lock(&m_statusMutex);
    return m_warningStatus;
}

QStringList RemoteDataProvider::getNewEvents()
{
    QMutexLocker lock(&m_eventMutex);
    QStringList events = m_events;
    m_events.clear();
    return events;
}

void RemoteDataProvider::cancelAlarm()
{
    // B版：通过MQTT发取消命令（后续Task集成MQTT后完善）
    qDebug("cancelAlarm via MQTT (placeholder)");
}
```

```c
#define SHM_OUTPUT_H

#include <stddef.h>

// 共享内存中的帧数据
struct shm_frame {

#endif // SHM_OUTPUT_H
```

- [ ] **Step 2: 创建实现文件**

写入 `src/shm_output.c`：

```c
#include "shm_output.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>

#define SHM_FRAME_NAME  "/edgewatcher_frame"
#define SHM_EVENTS_NAME "/edgewatcher_events"

int shm_output_init(shm_output_t *shm)
{
    if (!shm) return -1;

    memset(shm, 0, sizeof(*shm));

    // 创建帧共享内存
    size_t frame_size = sizeof(struct shm_frame);
    shm->shm_fd_frame = shm_open(SHM_FRAME_NAME,
                                  O_CREAT | O_RDWR, 0666);
    if (shm->shm_fd_frame < 0) {
        perror("shm_open(frame)");
        return -1;
    }
    if (ftruncate(shm->shm_fd_frame, frame_size) < 0) {
        perror("ftruncate(frame)");
        close(shm->shm_fd_frame);
        return -1;
    }
    shm->frame = mmap(NULL, frame_size,
                      PROT_READ | PROT_WRITE,
                      MAP_SHARED, shm->shm_fd_frame, 0);
    if (shm->frame == MAP_FAILED) {
        perror("mmap(frame)");
        close(shm->shm_fd_frame);
        return -1;
    }
    memset((void*)shm->frame, 0, frame_size);

    // 创建事件共享内存
    size_t events_size = sizeof(struct shm_events);
    shm->shm_fd_events = shm_open(SHM_EVENTS_NAME,
                                   O_CREAT | O_RDWR, 0666);
    if (shm->shm_fd_events < 0) {
        perror("shm_open(events)");
        munmap((void*)shm->frame, frame_size);
        close(shm->shm_fd_frame);
        return -1;
    }
    if (ftruncate(shm->shm_fd_events, events_size) < 0) {
        perror("ftruncate(events)");
        close(shm->shm_fd_events);
        munmap((void*)shm->frame, frame_size);
        close(shm->shm_fd_frame);
        return -1;
    }
    shm->events = mmap(NULL, events_size,
                       PROT_READ | PROT_WRITE,
                       MAP_SHARED, shm->shm_fd_events, 0);
    if (shm->events == MAP_FAILED) {
        perror("mmap(events)");
        close(shm->shm_fd_events);
        munmap((void*)shm->frame, frame_size);
        close(shm->shm_fd_frame);
        return -1;
    }
    memset((void*)shm->events, 0, events_size);

    printf("[shm_output] initialized\n");
    return 0;
}

void shm_output_write_frame(shm_output_t *shm,
                            const unsigned char *rgb,
                            int width, int height,
                            int frame_id)
{
    if (!shm || !shm->frame || !rgb) return;

    shm->frame->ready = 1;  // 正在写
    shm->frame->frame_id = frame_id;
    size_t row_len = (size_t)width * 3;
    size_t total = row_len * (size_t)height;
    memcpy((void*)shm->frame->rgb, rgb,
           total < sizeof(shm->frame->rgb) ? total : sizeof(shm->frame->rgb));
    shm->frame->ready = 0;  // 写完，QT可读
}

void shm_output_write_events(shm_output_t *shm,
                             int watcher, int motion, int warning)
{
    if (!shm || !shm->events) return;
    shm->events->watcher = watcher;
    shm->events->motion = motion;
    shm->events->warning = warning;
}

int shm_output_read_cancel(shm_output_t *shm)
{
    if (!shm || !shm->events) return 0;
    int val = shm->events->cancel;
    if (val) {
        shm->events->cancel = 0;  // 读取后清零
    }
    return val;
}

void shm_output_destroy(shm_output_t *shm)
{
    if (!shm) return;
    if (shm->events) {
        munmap((void*)shm->events, sizeof(struct shm_events));
    }
    if (shm->shm_fd_events >= 0) {
        close(shm->shm_fd_events);
        shm_unlink(SHM_EVENTS_NAME);
    }
    if (shm->frame) {
        munmap((void*)shm->frame, sizeof(struct shm_frame));
    }
    if (shm->shm_fd_frame >= 0) {
        close(shm->shm_fd_frame);
        shm_unlink(SHM_FRAME_NAME);
    }
    memset(shm, 0, sizeof(*shm));
    printf("[shm_output] destroyed\n");
}
```

```c
#include "tcp_stream.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

int tcp_stream_init(tcp_stream_t *ts, int port)
{
    if (!ts || port <= 0 || port > 65535) return -1;

    memset(ts, 0, sizeof(*ts));
    ts->port = port;
    ts->listen_fd = -1;
    ts->client_fd = -1;

    ts->listen_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (ts->listen_fd < 0) {
        perror("tcp socket");
        return -1;
    }

    // 端口复用
    int opt = 1;
    setsockopt(ts->listen_fd, SOL_SOCKET, SO_REUSEADDR,
               &opt, sizeof(opt));

    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons((uint16_t)port);

    if (bind(ts->listen_fd, (struct sockaddr*)&addr,
             sizeof(addr)) < 0) {
        perror("tcp bind");
        close(ts->listen_fd);
        return -1;
    }

    if (listen(ts->listen_fd, 1) < 0) {
        perror("tcp listen");
        close(ts->listen_fd);
        return -1;
    }

    printf("[tcp_stream] listening on port %d\n", port);
    return 0;
}

int tcp_stream_accept(tcp_stream_t *ts)
{
    if (!ts || ts->listen_fd < 0) return -1;

    printf("[tcp_stream] waiting for client...\n");
    ts->client_fd = accept(ts->listen_fd, NULL, NULL);
    if (ts->client_fd < 0) {
        perror("tcp accept");
        return -1;
    }

    printf("[tcp_stream] client connected\n");
    return 0;
}

int tcp_stream_send_frame(tcp_stream_t *ts,
                          const unsigned char *jpeg_data,
                          int len)
{
    if (!ts || ts->client_fd < 0 || !jpeg_data || len <= 0)
        return -1;

    // 协议: 4字节大端长度 + JPEG数据
    uint32_t net_len = htonl((uint32_t)len);
    if (send(ts->client_fd, &net_len, 4, MSG_NOSIGNAL) != 4)
        return -1;
    if (send(ts->client_fd, jpeg_data, len, MSG_NOSIGNAL) != len)
        return -1;

    return 0;
}

void tcp_stream_disconnect(tcp_stream_t *ts)
{
    if (!ts) return;
    if (ts->client_fd >= 0) {
        close(ts->client_fd);
        ts->client_fd = -1;
        printf("[tcp_stream] client disconnected\n");
    }
}

void tcp_stream_destroy(tcp_stream_t *ts)
{
    if (!ts) return;
    tcp_stream_disconnect(ts);
    if (ts->listen_fd >= 0) {
        close(ts->listen_fd);
        ts->listen_fd = -1;
    }
    printf("[tcp_stream] destroyed\n");
}
```

```cpp
bool LocalDataProvider::connect()
{
    // 打开帧共享内存
    m_shmFd_frame = shm_open(SHM_FRAME_NAME, O_RDWR, 0666);
    if (m_shmFd_frame < 0) {
        qWarning("shm_open(frame) failed, retrying with O_RDONLY");
        m_shmFd_frame = shm_open(SHM_FRAME_NAME, O_RDONLY, 0666);
        if (m_shmFd_frame < 0) {
            qCritical("Cannot open frame shared memory");
            return false;
        }
    }
    m_shmFrame = mmap(NULL, sizeof(ShmFrame),
                      PROT_READ, MAP_SHARED,
                      m_shmFd_frame, 0);
    if (m_shmFrame == MAP_FAILED) {
        qCritical("mmap(frame) failed");
        close(m_shmFd_frame);
        return false;
    }

    // 打开事件共享内存
    m_shmFd_events = shm_open(SHM_EVENTS_NAME, O_RDWR, 0666);
    if (m_shmFd_events < 0) {
        qWarning("shm_open(events) failed");
        munmap(m_shmFrame, sizeof(ShmFrame));
        close(m_shmFd_frame);
        return false;
    }
    m_shmEvents = mmap(NULL, sizeof(ShmEvents),
                       PROT_READ | PROT_WRITE, MAP_SHARED,
                       m_shmFd_events, 0);
    if (m_shmEvents == MAP_FAILED) {
        qCritical("mmap(events) failed");
        close(m_shmFd_events);
        munmap(m_shmFrame, sizeof(ShmFrame));
        close(m_shmFd_frame);
        return false;
    }

    qDebug("LocalDataProvider: connected via shared memory");
    return true;
}

void LocalDataProvider::disconnect()
{
    if (m_shmEvents) {
        munmap(m_shmEvents, sizeof(ShmEvents));
        m_shmEvents = nullptr;
    }
    if (m_shmFd_events >= 0) {
        close(m_shmFd_events);
        m_shmFd_events = -1;
    }
    if (m_shmFrame) {
        munmap(m_shmFrame, sizeof(ShmFrame));
        m_shmFrame = nullptr;
    }
    if (m_shmFd_frame >= 0) {
        close(m_shmFd_frame);
        m_shmFd_frame = -1;
    }
}
```
